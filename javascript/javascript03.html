01. 선언적 함수

직접 선언하는 함수입니다.

{
    function func(){
        document.write("실행 되었습니다.");  
    }
    func();   //먼저 호출 하여도 에러를 발생시키지 않습니다.
}
// var로 선언한 변수와 선언적 함수는 변수, 함수 선언 이전에 호출하여도 에러를 발생 시키징 않습니다. 이러한 자바스크립트 성질을 호이스팅(Hoistion) 이라고 합니다.

실행 되었습니다.

    }
    func();
